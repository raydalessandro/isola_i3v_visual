## §7.IMPLEMENTATION

### §7.1.PYTHON.FRAMEWORK

```python
"""
EAR-PERSONAGGI v2.0 - Character Generation Framework
"""

from dataclasses import dataclass
from typing import List, Dict, Tuple, Optional
from enum import Enum

# ============================================================================
# ENUMS
# ============================================================================

class Dimension(Enum):
    D1_LINEAR = 1
    D2_PLANAR = 2
    D3_VOLUMETRIC = 3
    D4_TEMPORAL = 4

class Attribute(Enum):
    A1_DISTINCTION = 1  # Δ
    A2_RELATION = 2     # ⇄
    A3_PROCESS = 3      # ⟳

class Axis(Enum):
    X1_FOUNDATIONAL = 1
    X2_RECURSIVE = 2
    X3_SYNTHETIC = 3

class Pole(Enum):
    POSITIVE = "+"
    NEGATIVE = "-"

# ============================================================================
# SIGMA CLASS
# ============================================================================

@dataclass(frozen=True)
class Sigma:
    """
    Σ_ijkp coordinate in ontological space
    """
    dimension: Dimension
    attribute: Attribute
    axis: Axis
    pole: Pole
    
    def __str__(self) -> str:
        d = self.dimension.value
        a = self.attribute.value
        x = self.axis.value
        p = self.pole.value
        return f"Σ₄{d}{a}{x}{p}"
    
    def distance_to(self, other: 'Sigma') -> int:
        """
        Calculate ontological distance to another Σ
        """
        d_dist = abs(self.dimension.value - other.dimension.value)
        a_dist = abs(self.attribute.value - other.attribute.value)
        x_dist = abs(self.axis.value - other.axis.value)
        p_dist = 0 if self.pole == other.pole else 1
        
        return d_dist + a_dist + x_dist + p_dist

# ============================================================================
# VOICE PATTERNS
# ============================================================================

@dataclass
class VoicePattern:
    """
    Voice characteristics derived from Σ
    """
    sigma: Sigma
    archetype_name: str
    speech_style: str
    rhythm: str
    typical_phrases: List[str]
    thought_process: str
    
    @classmethod
    def from_sigma(cls, sigma: Sigma) -> 'VoicePattern':
        """
        Derive voice pattern from Σ coordinate
        """
        # This would contain the full mapping
        # For now, simplified example
        
        if sigma == Sigma(
            Dimension.D2_PLANAR,
            Attribute.A3_PROCESS,
            Axis.X2_RECURSIVE,
            Pole.POSITIVE
        ):
            return cls(
                sigma=sigma,
                archetype_name="Lo Spiralista",
                speech_style="cyclical, evolving loops",
                rhythm="moderate to fast, returning patterns",
                typical_phrases=[
                    "It's like when...",
                    "and then you realize...",

                ],
